## List of pins used on the arduino uno

d2 = ultrasonic-sensor left echo
d3 = ultrasonic-sensor right echo
d4 = ultrasonic-sensor center echo
d5 = ultrasonic-sensor left trig
d6 = ultrasonic-sensor right trig
d7 = ultrasonic-sensor center trig

a4 = screen sda
a5 = screen scl

a4 = driver sda
a5 = driver scl
